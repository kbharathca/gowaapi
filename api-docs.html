Gowa API Documentation

All integration examples for the Cloudflare-hosted Gowa API.

Base URL

https://gowaapi.pages.dev/api

Authentication

Include headers:

X-Device-Id: official bharath
X-Api-Key: YOUR_PUBLIC_API_KEY

Endpoints

  • GET /devices
  • POST /send/message
  • POST /send/image
  • POST /send/video
  • POST /send/file

JavaScript (Browser)

const BASE = "https://gowaapi.pages.dev/api";
await fetch(`${BASE}/send/message`, {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "X-Device-Id": "official bharath",
    "X-Api-Key": "YOUR_PUBLIC_API_KEY"
  },
  body: JSON.stringify({phone:"919876543210@s.whatsapp.net",message:"Hello",is_forwarded:false})
});

Node.js

const fetch = require('node-fetch');
const BASE = "https://gowaapi.pages.dev/api";
await fetch(`${BASE}/send/image`, {
  method: "POST",
  headers: {"X-Device-Id":"official bharath","X-Api-Key":"YOUR_PUBLIC_API_KEY"},
  body: new URLSearchParams({phone:"919876543210@s.whatsapp.net",caption:"Pic",image_url:"https://example.com/pic.jpg"})
});

Python (requests)

import requests
BASE = "https://gowaapi.pages.dev/api"
headers = {"X-Device-Id":"official bharath","X-Api-Key":"YOUR_PUBLIC_API_KEY"}
resp = requests.post(f"{BASE}/send/file", headers=headers, data={"phone":"919876543210@s.whatsapp.net","caption":"Doc"}, files={"file":open('doc.pdf','rb')})
print(resp.json())

PHP (cURL)

$base = "https://gowaapi.pages.dev/api";
$ch = curl_init("$base/send/video");
curl_setopt_array($ch, [
  CURLOPT_POST => true,
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_HTTPHEADER => ["X-Device-Id: official bharath","X-Api-Key: YOUR_PUBLIC_API_KEY"],
  CURLOPT_POSTFIELDS => [
    "phone"=>"919876543210@s.whatsapp.net",
    "video_url"=>"https://example.com/video.mp4"
  ]
]);
$out = curl_exec($ch);
curl_close($ch);
echo $out;

Next.js API Route Proxy

// pages/api/gowa-proxy.js
export default async function handler(req,res){
  const proxyRes = await fetch(`https://gowaapi.pages.dev/api${req.url}`,{method:req.method,headers:req.headers,body:req.body});
  const data = await proxyRes.json();
  res.status(proxyRes.status).json(data);
}
